9. FILE HANDLING IN C 

    #include 
    #include 
        
    int main() {
            FILE *filePtr; // File pointer
            char dataToWrite[] = "This is a sample text written to the file.";
            char dataToRead[100];
        
        
            filePtr = fopen("example.txt", "w");
            if (filePtr == NULL) {
                printf("Error opening file for writing.\n");
                return 1;
            }
            fprintf(filePtr, "%s\n", dataToWrite);
            fclose(filePtr);
            printf("Data written to file successfully.\n");
        
        
            filePtr = fopen("example.txt", "r");
            if (filePtr == NULL) {
                printf("Error opening file for reading.\n");
                return 1;
            }
            printf("Reading data from file:\n");
            while (fgets(dataToRead, sizeof(dataToRead), filePtr) != NULL) {
                printf("%s", dataToRead);
            }
            fclose(filePtr);
        
            return 0;
    }

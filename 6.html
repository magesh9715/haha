7.STRUCTURE AND UNION IN C 

    #include
    #include
    struct Data
    {
        int i;
        float f;
        char str[20];

    };
    union DataUnion
    {
        int i;
        float f;
        char str[20];
    };
    int main(){
        union DataUnion MyUnion;
        struct Data Mystruct={10,22.5,"hello world"};
        printf("structure: %d,%f,%s \n",Mystruct.i,Mystruct.f,Mystruct.str);
        MyUnion.f=220.5;
        printf("union(float): %f \n",MyUnion.f);
        MyUnion.i=10;
        
        printf("union(int): %d \n",MyUnion.i);
        return 0;
        

    }
